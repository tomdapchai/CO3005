        func (p Pair) swap() Pair {
            return Pair{first: p.second, second: p.first};
        }
        
        func swap(a int, b int) {
            var temp = a;
            a := b;
            b := temp;
        }
        
        func (p Pair) swap() Pair {
            return Pair{first: p.first, second: p.second};
