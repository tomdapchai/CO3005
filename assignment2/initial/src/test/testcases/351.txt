
        func binarySearch(arr [100]int, x int) int {
            low := 0;
            high := len(arr) - 1;
            for low <= high {
                mid := low + (high - low) / 2;
                if (arr[mid] == x) {
                    return mid;
                }
                if (arr[mid] < x) {
                    low := mid + 1;
                } else {
                    high := mid - 1;
                }
            }
            return -1;
        }
