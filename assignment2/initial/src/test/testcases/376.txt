
        func lcs(X string, Y string, m int, n int) int {
            if (m == 0 || n == 0) {
                return 0;
            }
            if (X[m-1] == Y[n-1]) {
                return 1 + lcs(X, Y, m-1, n-1);
            } else {
                return max(lcs(X, Y, m, n-1), lcs(X, Y, m-1, n));
            }
        }
