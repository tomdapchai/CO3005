
        func selectionSort(arr [100]int) {
            n := len(arr);
            for i := 0; i < n - 1; i += 1 {
                min_idx := i;
                for j := i + 1; j < n; j += 1 {
                    if (arr[j] < arr[min_idx]) {
                        min_idx := j;
                    }
                }
                swap(arr, i, min_idx);
            }
        }
